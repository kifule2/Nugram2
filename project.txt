last_tap = models.DateTimeField(null=True, blank=True)

current_rate = models.FloatField(default=1.0)
session_start = models.DateTimeField(null=True, blank=True)
@property
def elapsed_hours(self):
return max(24 - self.elapsed_hours, 0)
@property
def remaining_time(self):
return min((self.elapsed_hours / 24) * 100, 100)
@property
def points_earned_today(self):
now = timezone.now()
if self.is_mining and self.session_start:
elapsed = (now - self.session_start).total_seconds() / 3600
if elapsed >= 24:
self.stop_mining()
self.is_mining = True
self.session_start = now
self.last_tap = now
self.save()
def stop_mining(self):
if self.is_mining and self.session_start:
elapsed = (timezone.now() - self.session_start).total_seconds() / 3600
if elapsed >= 24:
self.stop_mining()
return True
return False

@register.filter
def linkify(text):

return text
url_pattern = re.compile(
r'(https?://[^\s]+)|(www\.[^\s]+)|([a-zA-Z0-9.-]+\.[a-zA-Z]{2,}(?:/[^\s]*)?)'
)
def replace_url(match):
url = match.group(0)
if url.startswith('www.'):
full_url = 'https://' + url
elif not url.startswith('http'):
full_url = 'https://' + url
else:
full_url = url
display_url = url if len(url) <= 50 else url[:40] + '...'
return f'<a href="{full_url}" target="_blank" rel="noopener noreferrer" style="color:
result = url_pattern.sub(replace_url, text)
return mark_safe(result)

class Command(BaseCommand):

def handle(self, *args, **options):

return True, {'message': 'No keyword required', 'url': url}

}

try:
if streaming:
async with client.stream('GET', url, headers=headers) as response:

return False, {
'url': url,
'keyword': keyword,
'found': False,
'error': f'HTTP {response.status_code}'
}
bytes_read = 0
content = ''
keyword_lower = keyword.lower()
async for chunk in response.aiter_bytes():
chunk_str = chunk.decode('utf-8', errors='ignore')
content += chunk_str
bytes_read += len(chunk_str)
if keyword_lower in content.lower():
return True, {
'url': url,
'keyword': keyword,
'found': True,
'bytes_read': bytes_read,
'method': 'streaming'
}
if bytes_read >= MAX_STREAM_BYTES:
return False, {
'url': url,
'keyword': keyword,
'found': False,
'bytes_read': bytes_read,
'reason': f'Keyword not found in first {MAX_STREAM_BYTES} bytes',
'method': 'streaming'
}
return False, {
'url': url,
'keyword': keyword,
'found': False,
'bytes_read': bytes_read,
'reason': 'End of page reached',
'method': 'streaming'
}
else:
response = await client.get(url, headers=headers)
if response.status_code != 200:
return False, {
'url': url,
'keyword': keyword,
'found': False,
'error': f'HTTP {response.status_code}'
}

'url': url,
'keyword': keyword,

}

def sync_fetch():

}

comments.sort(key=lambda x: x.get('timestamp', 0), reverse=True)
fetch_info = {

'video_title': info.get('title', ''),
'video_id': info.get('id', '')
}
return comments, fetch_info
except Exception as e:

return None, {'error': str(e)}

result = await loop.run_in_executor(None, sync_fetch)
if result[0] is None:
return None
return result

recipient_profile = recipient.userprofile

sender_profile.token_balance -= amount
recipient_profile.token_balance += amount
transaction = Transaction.objects.create(
user=request.user,
amount=amount,
transaction_type='transfer',
status='completed',
recipient=recipient
)
sender_profile.save()
recipient_profile.save()
Notification.objects.create(
user=request.user,
message=f"You sent {amount} tokens to {recipient.username}.",
transaction=transaction
)
Notification.objects.create(
user=recipient,
message=f"You received {amount} tokens from {request.user.username}.",
transaction=transaction
)
messages.success(request, f"Transferred {amount} tokens to {username}")
else:
messages.error(request, "Insufficient balance")
except CustomUser.DoesNotExist:
messages.error(request, "User not found")
except ValueError:
